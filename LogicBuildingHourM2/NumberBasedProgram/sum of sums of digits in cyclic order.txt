import java.io.*;
import java.util.*;

class UserMainCode {
    public int sumOfSumsOfDigits(int input1) {
        String str = String.valueOf(input1);
        int totalSum = 0;

        for (int i = 0; i < str.length(); i++) {
            for (int j = i; j < str.length(); j++) {
                totalSum += str.charAt(j) - '0';
            }
        }

        return totalSum;
    }
}
